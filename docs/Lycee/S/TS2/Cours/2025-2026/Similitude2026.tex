\subsection*{\underline{1) Fonction dans $\mathbb{C}$ associée à une transformation du plan}}

\noindent
\begin{minipage}{0.5\textwidth}

    \[ z \to f(z) \]
\end{minipage}
\begin{minipage}{0.5\textwidth}
    \raggedleft

    \[ M(z) \to M'(z') \]

\vspace{1em}

\noindent \textbf{Si $f$ est une bijection de $\mathbb{C}$ dans $\mathbb{C}$, alors}

\vspace{1em}

\noindent \textbf{$F$ est la transformation du plan associée à la bijection complexe $f$ dans $\mathbb{C}$.}

\subsection*{\underline{3) Translation}}

\noindent \textbf{Toute transformation du plan d'écriture complexe $z' = z + b$, $b \in \mathbb{C}$, est une translation de vecteur $\vec{u}$ d'affixe $b$.}

\vspace{0.5em}
\subsection*{Démonstration : Écriture complexe d'une Translation}
\vspace{1em}

\begin{center}
    $t_{\vec{u}} : \mathcal{P} \to \mathcal{P}$ \\
    $M(z) \to t_{\vec{u}}(M) = M'(z')$
\end{center}

\noindent $t_{\vec{u}}(M) = M' \quad \iff \quad \vec{u} = \vec{MM'} \quad \iff \quad z_{\vec{u}} = z_{\vec{MM'}}$

\vspace{1em}

\noindent $z_{\vec{u}} = z_{\vec{MM'}} = z_{M'} - z_{M} \cdot z_{\vec{u}} = z_{M'} - z_{M}$. Or $z_{M'} = z'$, $z_{M} = z$.

\vspace{1em}

\noindent On pose : $z_{\vec{u}} = b$.

\vspace{1em}

\noindent On obtient : $b = z' - z \quad$ d'où $\quad z' = z + b$.

\vspace{1em}

\noindent \textbf{Toute écriture complexe de la forme : $z' = z + b$ est une translation de vecteur $\vec{u}$ d'affixe $b$.}

\subsection*{ \underline{Exercice d'application}}

\vspace{1em}

\noindent 1) Donner la nature de la transformation du plan d'écriture complexe : $z' = z - 3 + 2i$.

\noindent 2) Soit la translation de vecteur $\vec{u}$ d'affixe $-2 + i$. Donner son écriture complexe.

\vspace{1em}

\subsection*{ \underline{Solution}}

\vspace{1em}

\noindent 1) Donnons la nature de la transformation du plan d'écriture complexe : $z' = z - 3 + 2i$.

\noindent : $z' = z - 3 + 2i \quad , \quad b = -3 + 2i$.

\vspace{1em}

\noindent \fbox{\textbf{La transformation du plan est la translation de vecteur $\vec{u}$ d'affixe $-3 + 2i$.}}

\vspace{1em}

\noindent 2) Soit la translation de vecteur $\vec{u}$ d'affixe $-2 + i$. Donnons son écriture complexe.

\noindent Le vecteur $\vec{u}$ est d'affixe $-2 + i$ donc $b = -2 + i$.

\vspace{1em}

\noindent \fbox{\textbf{La translation de vecteur $\vec{u}$ d'affixe $-2 + i$ est d'écriture complexe : $z' = z - 2 + i$.}}

\section*{4) \underline{Homothétie}}

\noindent \textbf{Toute transformation du plan d'écriture complexe $z' = az + b$, $a \in \mathbb{R} \setminus \{0; 1\}$, $b \in \mathbb{C}$, est une homothétie de rapport $k = a$ et de centre $\Omega$ tel que $z_{\Omega} = \frac{b}{1 - a}$.}

\vspace{1em}

\subsection*{Démonstration : Écriture complexe d'une Homothétie}

\begin{center}
$
\begin{aligned}
h(\Omega, k) :& \mathcal{P}& \to \mathcal{P} \\
&M(z)& \to h(\Omega, k)(M) = M'(z')
\end{aligned}
$
\end{center}

\noindent $h(\Omega, k)(M) = M' \quad \iff \quad \overrightarrow{\Omega M'} = k \overrightarrow{\Omega M} \quad \iff \quad z_{\overrightarrow{\Omega M'}} = z_{k \overrightarrow{\Omega M}}$

\vspace{1em}

\noindent $z_{\overrightarrow{\Omega M'}} = z_{k \overrightarrow{\Omega M}} = k z_{\overrightarrow{\Omega M}} \quad \iff \quad z_{M'} - z_{\Omega} = k(z_M - z_{\Omega})$. Or $z_{M'} = z'$, $z_M = z$.

\vspace{1em}

\noindent On obtient : $z' - z_{\Omega} = k(z - z_{\Omega})$, \quad $z' - z_{\Omega} = kz - kz_{\Omega}$ \quad donc $z' = kz + (1 - k) z_{\Omega}$.

\vspace{1em}

\noindent Posons $k = a$ \quad et \quad $b = (1 - k) z_{\Omega}$ \quad donc $z' = az + b$.

\vspace{1em}

\noindent $k = a$ \quad donc $b = (1 - a) z_{\Omega}$ \quad donc \quad $z_{\Omega} = \frac{b}{1 - a}$.

\vspace{1em}

\noindent \textbf{Toute transformation du plan d'écriture complexe $z' = az + b$, $a \in \mathbb{R} \setminus \{0; 1\}$, $b \in \mathbb{C}$, est une homothétie de rapport $k = a$ et de centre $\Omega$ tel que $z_{\Omega} = \frac{b}{1 - a}$.}

\subsection*{ \underline{Exercice d'application}}

\vspace{1cm}

\begin{enumerate}
    \item \textit{Donner la nature de la transformation du plan d'écriture complexe : $z' = -\frac{1}{2}z + 3 - 6i$.}
    \item \textit{Soit l'homothétie de rapport -2 et de centre A d'affixe $-1 + i$. Donner son écriture complexe.}
\end{enumerate}

\subsection*{ \underline{Solution}}

\textbf{1) Donnons la nature de la transformation du plan d'écriture complexe : $z' = -\frac{1}{2} z + 3 - 6i$.}

$K = a = -\frac{1}{2} \quad b = 3 - 6i \quad . \quad z_{\Omega} = \frac{b}{1 - a} = \frac{3 - 6i}{1 - (-\frac{1}{2})} = \frac{3 - 6i}{\frac{3}{2}} = 2 - 4i \quad . \quad z_{\Omega} = 2 - 4i$

\begin{center}
\fbox{
    \begin{minipage}{0.95\textwidth}
    \textit{La transformation du plan d'écriture complexe : $z' = -\frac{1}{2} z + 3 - 6i$ est l'homothétie de rapport $-\frac{1}{2}$ et de centre $\Omega$ d'affixe $2 - 4i$.}
    \end{minipage}
}
\end{center}

\textbf{2) Soit l'homothétie de rapport -2 et de centre A d'affixe $-1 + i$. Donnons son écriture complexe.}

$k = a = -2 \quad , \quad z_A = -1 + i \quad , \quad b = (1 - a) z_A = (1 - (-2))(-1 + i) = -3 + 3i \quad , \quad b = -3 + 3i \quad .$

\begin{center}
\fbox{
    \begin{minipage}{0.95\textwidth}
    \textit{L'homothétie de rapport -2 et de centre A d'affixe $-1 + i$ est d'écriture complexe : $z' = -2 z - 3 + 3i$.}
    \end{minipage}
}
\end{center}

\section*{5) \underline{Rotation}}

\textbf{Toute transformation du plan d'écriture complexe $z' = a z + b$ , $a \in \mathbb{C} \setminus \mathbb{R}$ , $|a| = 1$ , $b \in \mathbb{C}$ , est une rotation d'angle $\theta = \text{arg } a \ (2\pi)$ et de centre $\Omega$ tel que $z_{\Omega} = \frac{b}{1 - a}$.}


 \textbf{Soit $z' = a z + b$ , $a \in \mathbb{C}$ , $|a| = 1$ , $b \in \mathbb{C}$ l'écriture complexe de la rotation d'angle $\theta$ et de centre $\Omega$ alors $a = e^{i\theta}$ et $b = (1 - a) z_{\Omega}$.}

\subsection*{Démonstration : Écriture complexe d'une rotation}

Soit $r$ une rotation de centre $\Omega$ d'affixe $z_{\Omega}$ et d'angle $\theta$. 
Soit $M$ un point du plan d'affixe $z$ et $M'$ son image par $r$ d'affixe $z'$.

\subsection*{1. Définition géométrique}
Par définition de la rotation $r(\Omega, \theta)$, le point $M'$ est l'image de $M$ si et seulement si :
\begin{itemize}
    \item $\Omega M' = \Omega M$ (conservation de la distance au centre)
    \item $(\overrightarrow{\Omega M}, \overrightarrow{\Omega M'}) \equiv \theta \pmod{2\pi}$ (mesure de l'angle orienté)
\end{itemize}

\subsection*{2. Traduction complexe}
Ces deux conditions se traduisent avec les affixes par :
\begin{enumerate}
    \item $|z' - z_{\Omega}| = |z - z_{\Omega}|$ ce qui implique $\left| \frac{z' - z_{\Omega}}{z - z_{\Omega}} \right| = 1$
    \item $\text{arg}\left( \frac{z' - z_{\Omega}}{z - z_{\Omega}} \right) \equiv \theta \pmod{2\pi}$
\end{enumerate}

Le nombre complexe $\frac{z' - z_{\Omega}}{z - z_{\Omega}}$ a donc pour module $1$ et pour argument $\theta$. 
D'après la forme exponentielle, nous avons :
\[ \frac{z' - z_{\Omega}}{z - z_{\Omega}} = e^{i\theta} \]

\subsection*{3. Passage à la forme $z' = az + b$}
En isolant $z'$, on obtient successivement :
\begin{itemize}
    \item $z' - z_{\Omega} = e^{i\theta}(z - z_{\Omega})$
    \item $z' = e^{i\theta} z - e^{i\theta} z_{\Omega} + z_{\Omega}$
    \item $z' = e^{i\theta} z + (1 - e^{i\theta}) z_{\Omega}$
\end{itemize}

\subsection*{Conclusion}
En posant $a = e^{i\theta}$ et $b = (1 - a) z_{\Omega}$, on retrouve bien la forme :
\[ z' = az + b \]
Où :
\begin{itemize}
    \item $|a| = |e^{i\theta}| = 1$
    \item $\theta = \text{arg}(a) \pmod{2\pi}$
    \item $z_{\Omega} = \frac{b}{1 - a}$ (affixe du point invariant)
\end{itemize}

\subsection*{Exercice d'application}

\begin{enumerate}
    \item Donner la nature de la transformation du plan d'écriture complexe : \(z' = \frac{1}{2}(1 - i\sqrt{3}) z - \sqrt{3} + i\).
    \item Soit la rotation d'angle \(-\frac{2\pi}{3}\) et de centre \(A\) d'affixe \(-2i\). Donner son écriture complexe.
\end{enumerate}

\subsection*{ \underline{Solution}}

\textbf{1) Donnons la nature de la transformation du plan d'écriture complexe : \(z' = \frac{1}{2}(1 - i\sqrt{3}) z - \sqrt{3} + i\).}

On a : \(a = \frac{1}{2}(1 - i\sqrt{3}) = \frac{1}{2} - i\frac{\sqrt{3}}{2}\) et \(b = -\sqrt{3} + i\).

\begin{itemize}
    \item \textbf{Module de \(a\) :} \(|a| = \sqrt{\left(\frac{1}{2}\right)^2 + \left(-\frac{\sqrt{3}}{2}\right)^2} = \sqrt{\frac{1}{4} + \frac{3}{4}} = 1\).
    \item \textbf{Argument de \(a\) :} \(\cos \theta = \frac{1}{2}\) et \(\sin \theta = -\frac{\sqrt{3}}{2}\), donc \(\theta = -\frac{\pi}{3} \ [2\pi]\).
    \item \textbf{Centre \(\Omega\) :} \(z_{\Omega} = \frac{b}{1 - a} = \frac{-\sqrt{3} + i}{1 - (\frac{1}{2} - i\frac{\sqrt{3}}{2})} = \frac{-\sqrt{3} + i}{\frac{1}{2} + i\frac{\sqrt{3}}{2}}\).
    
    En multipliant par l'expression conjuguée :
    \(z_{\Omega} = \frac{(-\sqrt{3} + i)(\frac{1}{2} - i\frac{\sqrt{3}}{2})}{\frac{1}{4} + \frac{3}{4}} = -\frac{\sqrt{3}}{2} + \frac{3}{2}i + \frac{1}{2}i + \frac{\sqrt{3}}{2} = 2i\).
\end{itemize}

\begin{center}
\fbox{
    \begin{minipage}{0.95\textwidth}
    \textbf{La transformation est la rotation d'angle \(-\frac{\pi}{3}\) et de centre \(\Omega\) d'affixe \(2i\).}
    \end{minipage}
}
\end{center}

\textbf{2) Soit la rotation d'angle \(-\frac{2\pi}{3}\) et de centre \(A\) d'affixe \(-2i\). Donnons son écriture complexe.}

On a : \(\theta = -\frac{2\pi}{3}\) et \(z_A = -2i\).
\begin{itemize}
    \item \(a = e^{i(-\frac{2\pi}{3})} = \cos(-\frac{2\pi}{3}) + i\sin(-\frac{2\pi}{3}) = -\frac{1}{2} - i\frac{\sqrt{3}}{2}\).
    \item \(b = (1 - a) z_A = \left(1 - (-\frac{1}{2} - i\frac{\sqrt{3}}{2})\right)(-2i) = \left(\frac{3}{2} + i\frac{\sqrt{3}}{2}\right)(-2i)\).
    \item \(b = -3i - i^2\sqrt{3} = \sqrt{3} - 3i\).
\end{itemize}

\begin{center}
\fbox{
    \begin{minipage}{0.95\textwidth}
    \textbf{L'écriture complexe de la rotation est : \(z' = \left(-\frac{1}{2} - i\frac{\sqrt{3}}{2}\right) z + \sqrt{3} - 3i\).}
    \end{minipage}
}
\end{center}
\section*{6) \underline{Similitude plane directe}}

\textbf{On appelle similitude plane directe toute transformation du plan d'écriture complexe :}
\[ z' = az + b \text{ , } a \in \mathbb{C} \setminus \mathbb{R} \text{ , } |a| \neq 1 \text{ , } b \in \mathbb{C} \text{ .} \]

\textbf{Les éléments caractéristiques de la similitude plane directe sont :}
\begin{itemize}
    \item[$\blacktriangleright$] \textbf{Son rapport : $\lambda = |a|$}
    \item[$\blacktriangleright$] \textbf{Son angle : $\theta = \text{arg}(a) \ (2\pi)$}
\end{itemize}

\begin{itemize}
    \item[$\blacktriangleright$] \textbf{Son centre : son centre est $\Omega$ tel que $z_{\Omega} = \frac{b}{1 - a}$ .}
\end{itemize}

\textbf{$\lambda$ , $\theta$ et $\Omega$ sont appelés les éléments caractéristiques de la similitude plane directe .}

\textbf{La similitude plane directe est souvent notée $S$ .}

\textbf{On dit $S$ est la similitude directe de rapport $\lambda$ , d'angle $\theta$ et de centre $\Omega$ .}

\begin{itemize}
    \item[$\blacktriangleright$] \textbf{Soit $z' = az + b$ , $a \in \mathbb{C} \setminus \mathbb{R}$ , $|a| \neq 1$ , $b \in \mathbb{C}$ l'écriture d'une similitude plane directe de rapport $\lambda$ , d'angle $\theta$ et de centre $\Omega$ alors $a = \lambda e^{i\theta}$ et $b = (1 - a) z_{\Omega}$ .}
\end{itemize}

\subsection*{Remarque}

\textbf{$z_{\Omega} = \frac{b}{1 - a} \iff b = (1 - a) z_{\Omega}$ .}

\textbf{$\lambda = |a|$ donc $|a| = \lambda$ , $\text{arg}(a) = \theta$ d'où $a = \lambda \times e^{i\theta} = \lambda e^{i\theta}$ .}

\textbf{Les deux écritures complexes suivantes sont équivalentes :}
\begin{enumerate}
    \item \textbf{Forme développée :} \( z' = az + b \)
    \item \textbf{Forme réduite :} \( z' - z_{\Omega} = \lambda e^{i\theta}(z - z_{\Omega}) \)
\end{enumerate}

\begin{comment}
\textbf{En effet, comme \( z_{\Omega} \) est le point invariant (\( z_{\Omega} = az_{\Omega} + b \)), on peut écrire :}
\[ z' - z_{\Omega} = (az + b) - (az_{\Omega} + b) \]
\[ z' - z_{\Omega} = a(z - z_{\Omega}) \]

\textbf{En remplaçant \( a \) par sa forme exponentielle \( a = \lambda e^{i\theta} \), on retrouve exactement la forme réduite. Cette dernière met en évidence que la similitude est la composée d'une rotation d'angle \( \theta \) et d'une homothétie de rapport \( \lambda \), toutes deux de même centre \( \Omega \).}
\end{comment}

\subsection*{ \underline{Exemple}}
\begin{enumerate}
    \item Soit la similitude directe $S$ telle que $z_A - z_B = 2 e^{i\frac{\pi}{4}}(z_C - z_B)$.
    
    On a $S(C) = A$ , $\lambda = 2$ , $\theta = \frac{\pi}{4}$.
    
    $S$ est la similitude directe de rapport 2, d'angle $\frac{\pi}{4}$ et son centre est le point $B$.
    
    \item Soit la similitude directe $S$ telle que $z_E - z_D = 5 e^{i\frac{2\pi}{3}}(z_F - z_D)$.
    
    On a $S(F) = E$ , $\lambda = 5$ , $\theta = \frac{2\pi}{3}$.
    
    $S$ est la similitude directe de rapport 5, d'angle $\frac{2\pi}{3}$ et son centre est le point $D$.
\end{enumerate}

\begin{enumerate}
    \item Donner les éléments caractéristiques de la similitude plane directe $S$ d'écriture complexe :
    \[ z' = (1 + i) z - 2i \text{ .} \]
    \item Donner l'écriture complexe de la similitude plane directe de rapport $2$ , d'angle $-\frac{\pi}{3}$ et de centre $\Omega$ d'affixe $1 + i$ .
\end{enumerate}
\subsection*{ \underline{Solution}}

\textbf{1) Déterminons les éléments caractéristiques de la similitude plane directe \(S\) d'écriture complexe : \(z' = (1 + i) z - 2i\).}

On a : \(a = 1 + i\) et \(b = -2i\).

\begin{itemize}
    \item \textbf{Rapport \(\lambda\) :} \(\lambda = |a| = |1 + i| = \sqrt{1^2 + 1^2} = \sqrt{2}\).
    \item \textbf{Angle \(\theta\) :} \(\cos \theta = \frac{1}{\sqrt{2}} = \frac{\sqrt{2}}{2}\) et \(\sin \theta = \frac{1}{\sqrt{2}} = \frac{\sqrt{2}}{2}\), donc \(\theta = \frac{\pi}{4} \ [2\pi]\).
    \item \textbf{Centre \(\Omega\) :} \(z_{\Omega} = \frac{b}{1 - a} = \frac{-2i}{1 - (1 + i)} = \frac{-2i}{-i} = 2\).
\end{itemize}

\begin{center}
\fbox{
    \begin{minipage}{0.95\textwidth}
    \textbf{La similitude plane directe \(S\) a pour rapport \(\sqrt{2}\), pour angle \(\frac{\pi}{4}\) et pour centre \(\Omega\) d'affixe \(2\).}
    \end{minipage}
}
\end{center}

\textbf{2) Donnons l'écriture complexe de la similitude plane directe de rapport 2, d'angle \(-\frac{\pi}{3}\) et de centre \(\Omega\) d'affixe \(1 + i\).}

On a : \(\lambda = 2\), \(\theta = -\frac{\pi}{3}\) et \(z_{\Omega} = 1 + i\).
\begin{itemize}
    \item \textbf{Calcul de \(a\) :} \(a = \lambda e^{i\theta} = 2 \left(\cos(-\frac{\pi}{3}) + i\sin(-\frac{\pi}{3})\right) = 2 \left(\frac{1}{2} - i\frac{\sqrt{3}}{2}\right) = 1 - i\sqrt{3}\).
    \item \textbf{Calcul de \(b\) :} \(b = (1 - a) z_{\Omega} = (1 - (1 - i\sqrt{3}))(1 + i) = (i\sqrt{3})(1 + i) = i\sqrt{3} + i^2\sqrt{3} = -\sqrt{3} + i\sqrt{3}\).
\end{itemize}

\begin{center}
\fbox{
    \begin{minipage}{0.95\textwidth}
    \textbf{L'écriture complexe de la similitude est : \(z' = (1 - i\sqrt{3}) z - \sqrt{3} + i\sqrt{3}\).}
    \end{minipage}
}
\end{center}

        $a \in \mathbb{R} \setminus \{0 ; 1\}$ & \textbf{Homothétie} &

\subsection*{ Comment identifier la transformation ?}

\begin{tcolorbox}[colback=blue!5,colframe=blue!75!black,title=Méthode de reconnaissance]
Regardez attentivement le coefficient \textbf{$a$} :
\begin{enumerate}
    \item \textbf{Si $a = 1$} : C'est une \textbf{Translation} de vecteur d'affixe $b$.
    \item \textbf{Si $a$ est un nombre réel} ($a \in \mathbb{R} \setminus \{0, 1\}$) : C'est une \textbf{Homothétie}.
    \item \textbf{Si $a$ est complexe ($a \notin \mathbb{R}$) et $|a| = 1$} : C'est une \textbf{Rotation}.
    \item \textbf{Si $a$ est complexe ($a \notin \mathbb{R}$) et $|a| \neq 1$} : C'est une \textbf{Similitude plane directe}.
\end{enumerate}
\end{tcolorbox}

\subsection*{ \underline{Propriété}}
\textbf{Une similitude plane directe est soit une translation, soit une homothétie, soit une rotation, soit une composition commutative de rotation et d'homothétie de même centre.}
\section*{9) Détermination d'une similitude à partir de ses éléments caractéristiques}
\begin{enumerate}
\item  À partir de deux points et de leurs images

Soit la similitude plane directe $S$ telle que $S(A) = A'$ et $S(B) = B'$.

La similitude plane directe $S$ est d'écriture complexe : $z' = az + b$ avec $a = \frac{z_{A'} - z_{B'}}{z_A - z_B}$ et

$b = z_{A'} - az_A$.

\item À partir de son centre, d'un point et son image

Soit la similitude plane directe $S$ telle que $S(A) = A'$ et $S(\Omega) = \Omega$.

La similitude plane directe $S$ est d'écriture complexe : $z' = az + b$ avec $a = \frac{z_{A'} - z_{\Omega}}{z_A - z_{\Omega}}$ et

$b = z_{\Omega} - az_{\Omega}$.
\end{enumerate}

\section*{Démonstration : Détermination par le centre, un point et son image}
\begin{enumerate}
\item
$z' = az + b$ , $S(A) = A'$ et $S(B) = B'$ .

$S(A) = A'$ donc $z_{A'} = az_A + b \ (1)$ , $S(B) = B'$ donc $z_{B'} = az_B + b \ (2)$

En faisant la différence de $(1)$ et $(2)$, on a $z_{A'} - z_{B'} = a(z_A - z_B)$ d'où $a = \frac{z_{A'} - z_{B'}}{z_A - z_B}$

$z_{A'} = az_A + b$ donc $b = z_{A'} - az_A$ .

Donc $z' = az + b$ avec $a = \frac{z_{A'} - z_{B'}}{z_A - z_B}$ et $b = z_{A'} - az_A$ .

\item

$z' = az + b$ , $S(A) = A'$ et $S(\Omega) = \Omega$ .

$S(A) = A'$ donc $z_{A'} = az_A + b \ (1)$ , $S(\Omega) = \Omega$ donc $z_{\Omega} = az_{\Omega} + b \ (2)$

En faisant la différence de $(1)$ et $(2)$, on a $z_{A'} - z_{\Omega} = a(z_A - z_{\Omega})$ d'où $a = \frac{z_{A'} - z_{\Omega}}{z_A - z_{\Omega}}$

$z_{\Omega} = az_{\Omega} + b$ donc $b = z_{\Omega} - az_{\Omega}$ .

Donc $z' = az + b$ avec $a = \frac{z_{A'} - z_{\Omega}}{z_A - z_{\Omega}}$ et $b = z_{\Omega} - az_{\Omega}$ .
\end{enumerate}

\section*{Exercice d'application : Similitudes Planes Directes}

Le plan complexe est rapporté à un repère orthonormé direct $(O; \vec{u}, \vec{v})$.

Soit $s$ la similitude plane directe qui transforme le point $A$ d'affixe $z_A = 1 + i$ en $A'$ d'affixe $z_{A'} = 2 + 5i$, et le point $B$ d'affixe $z_B = i$ en $B'$ d'affixe $z_{B'} = 4 + 3i$.
\begin{enumerate}
    \item Déterminer le coefficient complexe $a$ de cette similitude.
    \item En déduire la valeur de $b$ et donner l'écriture complexe de $s$.
\end{enumerate}

\section*{Correction détaillée}

\begin{enumerate}
    \item \textbf{Calcul de $a$ :}
    D'après la formule $a = \frac{z_{A'} - z_{B'}}{z_A - z_B}$ :
    \[ a = \frac{(2 + 5i) - (4 + 3i)}{(1 + i) - i} = \frac{-2 + 2i}{1} = -2 + 2i \]
    
    \item \textbf{Calcul de $b$ :}
    En utilisant $b = z_{A'} - az_A$ :
    \[ b = (2 + 5i) - (-2 + 2i)(1 + i) = (2 + 5i) - (-2 - 2i + 2i + 2i^2) \]
    \[ b = (2 + 5i) - (-2 - 2) = (2 + 5i) - (-4) = 6 + 5i \]
    L'écriture complexe est : \textbf{$z' = (-2 + 2i)z + 6 + 5i$}.
\end{enumerate}
