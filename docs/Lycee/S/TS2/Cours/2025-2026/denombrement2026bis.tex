\section*{\underline{Dénombrement}}

\subsection*{I) Cardinal d'un ensemble fini}

\subsubsection*{1) Définition}
\fbox{
\begin{minipage}{0.5\textwidth}
Soit $E$ un ensemble fini non vide. Le cardinal de $E$ est le nombre d'éléments contenus dans $E$. \\
On le note : {\color{red} $Card(E)$}
\end{minipage}
}

\paragraph{Exemples :}
\begin{itemize}
    \item $card \{a\} = 1$ ; $card \{a; b\} = 2$
    \item $card D_{14} = 4$ ; $\dots$
\end{itemize}

\subsubsection*{2) Remarques}
\begin{itemize}
    \item $card \emptyset = 0$
    \item $A \subset B \implies card A \leqslant card B$
\end{itemize}

\subsubsection*{3) Résultat}
$A, B$ ensembles finis \\
{\Large $card(A \cup B) = {\color{red} card A + card B - card(A \cap B)}$}

        \node at (0.4, 0.5) {\small a};

    \end{scope}
\end{tikzpicture}
\end{center}

\vspace{1em}

\begin{itemize}
    \item \textbf{Pièce de Monnaie :} $\left\{ \text{Pile} ; \text{Face} \right\}$
    \item \textbf{Dé cubique :} $1 ; 2 ; 3 ; 4 ; 5 ; 6$
    \item \textbf{Dé tétraédrique :} $a ; b ; c ; d$
\end{itemize}

\vspace{1em}

\begin{enumerate}
    \item Quel est le nombre total de cas possibles ?
    \item Quel est le nombre de possibilités sachant que le ``a'' ne peut être mis dans une même combinaison avec un nombre pair ?
\end{enumerate}

\vspace{2em}
\hrule
\vspace{1em}

\section*{\underline{Solution (Indications)}}

\begin{enumerate}
    \item \textbf{Nombre total} : 
    D'après la loi produit : $2 \times 6 \times 4 = \mathbf{48}$ cas.

    \item \textbf{Avec contrainte} : 
    Il faut séparer le problème en deux cas pour le dé tétraédrique :
    \begin{itemize}
        \item Si on tire $\{b, c, d\}$ (3 choix) : Pas de contrainte sur le dé cubique ($3 \times 2 \times 6 = 36$).
        \item Si on tire $\{a\}$ (1 choix) : Le dé cubique doit être impair $\{1, 3, 5\}$ (3 choix). Soit $1 \times 2 \times 3 = 6$.
        \item \textbf{Total} : $36 + 6 = \mathbf{42}$ possibilités.
    \end{itemize}
\end{enumerate}

\vspace{2em}
\hrule
\vspace{1em}

\noindent \textbf{Solution :}
\begin{enumerate}
    \item Selon le principe multiplicatif : $2 \times 6 \times 4 = \mathbf{48}$ cas possibles.
    \item Pour le dé tétraédrique, il y a deux situations :
    \begin{itemize}
        \item Si on obtient $\{b, c, d\}$ (3 choix), le dé cubique peut être n'importe quoi (6 choix) : $2 \times 6 \times 3 = 36$.
        \item Si on obtient $\{a\}$ (1 choix), le dé cubique doit être impair $\{1, 3, 5\}$ (3 choix) : $2 \times 3 \times 1 = 6$.
        \item \textbf{Total} : $36 + 6 = \mathbf{42}$ possibilités.
    \end{itemize}
\end{enumerate}

\vspace{2em}
\hrule
