\question \textbf{Énoncer le théorème des valeurs intermédiaires (TVI).}\\[0.3cm]
\underline{\color{red}Soit $f$ une fonction \textbf{continue} sur un intervalle $[a, b]$.}\\[0.3cm]
\underline{\color{red}Pour tout nombre réel $k$ compris entre $f(a)$ et $f(b)$ (c'est-à-dire $k \in [\min(f(a), f(b)), \max(f(a), f(b))]$),}\\[0.3cm]
\underline{\color{red}l'équation $f(x) = k$ admet \textbf{au moins une} solution $x_0$ dans l'intervalle $[a, b]$.}\\

\vspace{0.3cm}

\question \textbf{Énoncer le théorème d’existence et d’unicité d’une solution (Corollaire du TVI).}\\[0.3cm]
\underline{\color{red}Soit $f$ une fonction \textbf{continue} et \textbf{strictement monotone} sur un intervalle $[a, b]$.}\\[0.3cm]
\underline{\color{red}Pour tout nombre réel $k$ compris entre $f(a)$ et $f(b)$,}\\[0.3cm]
\underline{\color{red}l'équation $f(x) = k$ admet une \textbf{unique} solution $\alpha$ dans l'intervalle $[a, b]$.}\\

\vspace{0.3cm}
