    \item Discuter suivant les valeurs du paramètre réel $m$ l'existence et le signe des solutions de l'équation $(E_m)$
    \item Déterminer $m$ pour que les racines $x'$ et $x''$ de $(E_m)$ vérifient l'égalité : $(2x' - 1)(2x'' - 1) = 6$
    \item Déterminer une relation entre $x'$ et $x''$ indépendante de $m$.
    \item Déterminer $m$ pour que l'équation $(E_m)$ admette :
    \begin{enumerate}
        \item Deux solutions de signes positifs.
        \item Deux solutions de signes contraires.
    \end{enumerate}
    \item Former une équation du second degré ayant pour solutions $X' = 3x' - 2$ et $X'' = 3x'' - 2$
\end{enumerate}

\fbox{\textbf{\exo}}

Soit $(E) : x^4 + 10x^3 + 26x^2 + 10x + 1 = 0$
\begin{enumerate}
    \item Montrer que 0 n'est pas solution de $(E)$.
    \item En déduire que $(E)$ a les mêmes solutions que $(E') : x^2 + 10x + 26 + \frac{10}{x} + \frac{1}{x^2} = 0$
    \item On pose $X = x + \frac{1}{x}$
    \begin{enumerate}
        \item Montrer que $x^2 + \frac{1}{x^2} = X^2 - 2$.
        \item Montrer que si $x$ est solution de $(E')$ alors $X$ est solution de l'équation $(E'') : X^2 + 10X + 24 = 0$
    \end{enumerate}
    \item Résoudre $(E'')$, puis en déduire les solutions de $(E)$.
    \item Utiliser la même méthode pour résoudre $20x^4 - 19x^3 - 402x^2 - 19x + 20 = 0$
\end{enumerate}
