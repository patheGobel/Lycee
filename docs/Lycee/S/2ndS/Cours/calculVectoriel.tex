\section*{I - Généralités}
\subsection*{1- Définitions}

\textit{\color{red}Définition 1 :}\\
On appelle \textbf{plan vectoriel} l'ensemble des vecteurs du plan.

\vspace{1.5em}

\textit{\color{red}Définition 2 :}\\
Soient $\vec{u}$ un vecteur du plan vectoriel et $(A, B)$ un couple de points. On dit que le couple $(A, B)$ est un \textbf{représentant} du vecteur $\vec{u}$ si et seulement si $\overrightarrow{AB} = \vec{u}$.

\begin{center}
    \begin{tikzpicture}[>=stealth, thick]
        % Premier vecteur (u)
        \draw[->] (0,0) -- (1.5,1.5) node[midway, right] {\color{red} $\vec{u}$};
        
        % Deuxième vecteur (AB) décalé vers la droite
        % On utilise les mêmes coordonnées relatives pour qu'ils soient parallèles
        \draw[->] (2,0) node[below left] {\color{red} $A$} -- (3.5,1.5) node[below right] {\color{red} $B$};
    \end{tikzpicture}
\end{center}
\textit{\color{red}Remarque 1 :}\\
Tout vecteur du plan a une infinité de représentants.
\begin{center}
    \begin{tikzpicture}[>=stealth, thick]
        % Vecteurs parallèles
        \draw[->] (0,1.2) node[above] {$E$} -- (2,1.2) node[right] {$F$};
        \draw[->] (0,0.8) -- (2,0.8) node[right] {$\vec{u}$};
        \draw[->] (0,0.4) node[left] {$A$} -- (2,0.4) node[right] {$B$};
        \draw[->] (0,0) node[left] {$C$} -- (2,0) node[right] {$D$};
    \end{tikzpicture}
    \quad $\vec{u} = \overrightarrow{AB} = \overrightarrow{CD} = \overrightarrow{EF} = ...$
\end{center}

\vspace{1em}

\textit{\color{red}Remarque 2 : (Caractéristiques d'un vecteur).}\\
Un vecteur est caractérisé par un \textcolor{red}{sens}, une \textcolor{red}{direction} et une \textcolor{red}{norme} :

\begin{itemize}
    \item[$*_{1}$] La \textcolor{red}{direction} est celle de la "droite" dans laquelle est inclus le vecteur.
    \item[$*_{2}$] Le \textcolor{red}{sens} est donné par l'orientation du segment dans lequel est inclus le vecteur ("vers la gauche" ou bien "vers la droite").
    \item[$*_{3}$] La \textcolor{red}{norme} correspond à la longueur du segment sur lequel est inclus le vecteur.
\end{itemize}

Ainsi, on en déduit la propriété suivante :

\subsection*{\underline{\color{red}Propriété :}}
\begin{enumerate}
    \item[i)] $\overrightarrow{AB} = \vec{0} \iff A = B \quad (\text{donc } \overrightarrow{AA} = \vec{0})$
    \item[ii)] $\overrightarrow{AB} = -\overrightarrow{BA}$
\end{enumerate}

\section*{2- Propriétés}
\subsection*{\underline{\color{red}Activité}}
Soient $O$ un point du plan et $\vec{u}$ un vecteur du plan vectoriel.
\begin{enumerate}
    \item[i)] Construire un point $A$ tel que $\overrightarrow{OA} = \vec{u}$. ($A \neq O$)
    \item[ii)] Construire un point $B$ tel que $\overrightarrow{OB} = \vec{u}$. ($B \neq O$)
    \item[iii)] Que peut-on dire des points $A$ et $B$ ?
\end{enumerate}

\subsection*{\underline{\color{red}Correction}}
\begin{enumerate}
    \item[i)] Construisons le point $A$.
    \begin{center}
        \begin{tikzpicture}[>=stealth, thick]
            \draw[->] (0,1) node[above] {$O$} -- (2,1) node[above] {$A$};
            \node at (2,0.7) {$B$};
            \draw[->] (0.5,0.2) -- (2.5,0.2) node[midway, below] {$\vec{u}$};
        \end{tikzpicture}
    \end{center}
    \item[ii)] Construisons le point $B$ (voir la question i).
    \item[iii)] On peut dire que les points $A$ et $B$ sont \textbf{confondus}.
\end{enumerate}
\subsection*{\underline{Propriété 1 : (Propriété fondamentale)}}
Pour tout point $O$ et pour tout vecteur $\vec{u}$, il existe un et un seul (unique) point $M$ tel que $\overrightarrow{OM} = \vec{u}$.

\subsubsection*{\underline{Preuve :}}
$\overrightarrow{OM} = \vec{u}$ signifie que $M$ est l'image de $O$ par la translation du vecteur $\vec{u}$. D'où le point $M$ est l'unique point tel que $\overrightarrow{OM} = \vec{u}$.

\vspace{1em}

\subsection*{\underline{Propriété 2 :}}
Soient $A, B, C$ et $D$ des points du plan. Alors les 3 énoncés suivants sont équivalents :
\begin{enumerate}
    \item[i)] $\overrightarrow{AB} = \overrightarrow{DC}$
    \item[ii)] $ABDC$ est un parallélogramme.
    \item[iii)] Les segments $[AC]$ et $[BD]$ ont le même milieu.
\end{enumerate}

\subsection*{3- \underline{\color{red}Norme d'un vecteur}}

\subsubsection*{\underline{\color{red}Définition :}}
Soit $\vec{u}$ un vecteur du plan de représentant $(A, B)$, ($\overrightarrow{AB} = \vec{u}$).\\
On appelle \textcolor{red}{norme} du vecteur notée $||\vec{u}||$ la distance $AB$ :
$ ||\vec{u}|| = AB $

\subsubsection*{\underline{\color{red}Propriété :}}
Soit $\vec{u}$ un vecteur du plan vectoriel :
\begin{enumerate}
    \item[i)] $||\vec{u}|| = 0 \iff \vec{u} = \vec{0}$
    \item[ii)] $||-\vec{u}|| = ||\vec{u}||$
\end{enumerate}
\subsubsection*{\underline{\color{red}Preuve :}}
\begin{enumerate}
    \item[i)] Montrons que $||\vec{u}|| = 0 \iff \vec{u} = \vec{0}$
    \begin{itemize}
        \item[$*_{1}$] $\vec{u} = \vec{0} \implies \vec{u} = \overrightarrow{AA}$ or $||\overrightarrow{AA}|| = 0$ d'où $||\vec{u}|| = 0$.
        \item[$*_{2}$] $||\vec{u}|| = 0$ donc si $(A, B)$ est un représentant de $\vec{u}$ ($\overrightarrow{AB} = \vec{u}$) alors $AB = 0$. Donc $A = B \iff \overrightarrow{AB} = \vec{0}$. D'où $\vec{u} = \vec{0}$.
    \end{itemize}
    \item[ii)] Montrons que $||-\vec{u}|| = ||\vec{u}||$ \\
    $(A, B)$ est un représentant du vecteur $\vec{u}$ équivaut à $(B, A)$ est un représentant du vecteur $-\vec{u}$.\\
    Donc $||\vec{u}|| = AB$ et $||-\vec{u}|| = BA$.\\
    Comme $AB = BA$ alors on en déduit que $||\vec{u}|| = ||-\vec{u}||$.
\end{enumerate}

\begin{comment}
Cette section explique que l'égalité des normes n'implique 
pas nécessairement l'égalité des vecteurs.
\end{comment}

\subsection*{\underline{\color{red}Remarque 1 :}}
Deux vecteurs de même norme ne sont pas nécessairement égaux.

\vspace{0.5em}

\noindent \underline{\color{red}Par exemple :} Dans $\mathbb{R}^2$, on prend :
$\vec{u} = \begin{pmatrix} 2 \\ 1 \end{pmatrix}$ et $\vec{v} = \begin{pmatrix} 1 \\ 2 \end{pmatrix}$

\vspace{0.5em}

\noindent On a :
\begin{align*}
||\vec{u}|| &= \sqrt{2^2 + 1^2} = \sqrt{5} \\
||\vec{v}|| &= \sqrt{1^2 + 2^2} = \sqrt{5}
\end{align*}

\noindent Donc $||\vec{u}|| = ||\vec{v}||$ mais $\vec{u} \neq \vec{v}$.

\begin{comment}
Cette section présente la conclusion sur la comparaison des vecteurs 
par leurs normes ainsi que les propriétés liées à la somme de vecteurs.
\end{comment}

\noindent \underline{Conclusion} : \\
$\|\vec{u}\| = \|\vec{v}\| \not\implies \vec{u} = \vec{v}$

\subsection*{\underline{\color{red}Remarque 2 :}}
Soient $\vec{u}$ et $\vec{v}$ deux vecteurs.
\begin{enumerate}
    \item[i)] \textbf{En général}, on a : \\
    $\|\vec{u} + \vec{v}\| \leq \|\vec{u}\| + \|\vec{v}\|$
    
    \item[ii)] \textbf{Par contre}, si $\vec{u}$ et $\vec{v}$ sont orthogonaux ($\vec{u} \perp \vec{v}$), alors : \\
    $\|\vec{u} + \vec{v}\|^2 = \|\vec{u}\|^2 + \|\vec{v}\|^2$
\end{enumerate}

\noindent Cette propriété est le \underline{théorème de Pythagore}.

\noindent \underline{\color{red}Exemple :} \\
i) $\vec{u} = \begin{pmatrix} 2 \\ -1 \end{pmatrix}$ et $\vec{v} = \begin{pmatrix} 2 \\ 1 \end{pmatrix}$ \\
On a : \\
$\vec{u} + \vec{v} = \begin{pmatrix} 4 \\ 0 \end{pmatrix}$ \\
$\|\vec{u} + \vec{v}\| = \sqrt{4^2 + 0^2} = 4 \quad (*)$ \\
$\begin{cases} 
\|\vec{u}\| = \sqrt{2^2 + (-1)^2} = \sqrt{5} \\ 
\|\vec{v}\| = \sqrt{2^2 + 1^2} = \sqrt{5} 
\end{cases}$ \\
$\|\vec{u}\| + \|\vec{v}\| = 2\sqrt{5} \quad (**)$ \\
D'après $(*)$ et $(**)$, on a : \\
$\|\vec{u} + \vec{v}\| < \|\vec{u}\| + \|\vec{v}\|$
\begin{comment}
Exemple ii) : Vérification du théorème de Pythagore pour deux vecteurs
orthogonaux u et v dans R^2.
\end{comment}

\noindent ii) $\vec{u} = \begin{pmatrix} 2 \\ 1 \end{pmatrix}$ et $\vec{v} = \begin{pmatrix} -1 \\ 2 \end{pmatrix}$ \\
On a : \\
$x x' + y y' = 2(-1) + 1(2) = 0$ \\
donc $\vec{u} \perp \vec{v}$ \\
$\vec{u} + \vec{v} = \begin{pmatrix} 1 \\ 3 \end{pmatrix}$ \\
$\|\vec{u} + \vec{v}\| = \sqrt{1^2 + 3^2} = \sqrt{10}$ \\
Donc \\
$\|\vec{u} + \vec{v}\|^2 = 10 \quad (1)$ \\
$\begin{cases} 
\|\vec{u}\| = \sqrt{5} \\ 
\|\vec{v}\| = \sqrt{5} 
\end{cases}$ \\
Donc, $\|\vec{u}\|^2 + \|\vec{v}\|^2 = 10 \quad (2)$ \\
D'après $(1)$ et $(2)$, on a : \\
$\|\vec{u} + \vec{v}\|^2 = \|\vec{u}\|^2 + \|\vec{v}\|^2$

\section*{\underline{\uppercase{\uppercase{II - Calcul vectoriel}}}}

\subsection*{1- \underline{\color{red}Relation de Chasles}}

\noindent Pour tous points $A$, $B$ et $C$, on a :
\begin{center}
    \color{red}
    \framebox{
        \color{black}
        $\overrightarrow{AB} + \overrightarrow{BC} = \overrightarrow{AC}$
    }
\end{center}
\noindent Cette relation est appelée \underline{relation de Chasles}.

\vspace{1.5em}

\subsection*{\underline{\color{red}Application}}

\noindent \textbf{i)} Soient $A$, $B$, $C$ et $M$ des points du plan tels que $\overrightarrow{AM} = \frac{1}{3} \overrightarrow{BC}$ \\
et $\overrightarrow{BM} = \dfrac{2}{3} \overrightarrow{BC}$. Montrer que $\overrightarrow{AB} = -\dfrac{1}{3} \overrightarrow{BC}$

\vspace{1em}

\noindent \textbf{ii)} Simplifier l'expression de $\vec{u} = \overrightarrow{AB} + 2\overrightarrow{BC} + 3\overrightarrow{CA} + \overrightarrow{CB}$

\section*{\underline{\color{red}Solution}}

\noindent \textbf{i) Montrons que $\overrightarrow{AB} = -\frac{1}{3} \overrightarrow{BC}$}

\vspace{1em}

\noindent $\overrightarrow{AB} = \overrightarrow{AM} + \overrightarrow{MB}$ \\
or $\overrightarrow{AM} = \frac{1}{3} \overrightarrow{BC}$ \quad et \quad $\overrightarrow{BM} = \frac{2}{3} \overrightarrow{BC} \implies \overrightarrow{MB} = -\frac{2}{3} \overrightarrow{BC}$

\vspace{1em}

\noindent Donc $\overrightarrow{AB} = \frac{1}{3} \overrightarrow{BC} - \frac{2}{3} \overrightarrow{BC}$

\vspace{1em}

\begin{center}
    \framebox{
        $\overrightarrow{AB} = -\frac{1}{3} \overrightarrow{BC}$
    }
\end{center}

\vspace{2em}

\noindent \textbf{ii) Simplifions l'expression :}

\vspace{1em}

\noindent $\vec{u} = \overrightarrow{AB} + 2\overrightarrow{BC} + 3\overrightarrow{CA} + \overrightarrow{CB}$ \\
$\vec{u} = \underbrace{\overrightarrow{AB} + \overrightarrow{BC}}_{\overrightarrow{AC}} + \overrightarrow{BC} + \overrightarrow{CA} + \overrightarrow{CA} + \overrightarrow{CA} + \overrightarrow{CB}$ \\
$\vec{u} = \overrightarrow{AC} + \overrightarrow{BA} + \overrightarrow{BC} + \overrightarrow{CA} + \overrightarrow{CA} + \overrightarrow{CB}$ \\
$\vec{u} = \overrightarrow{AC} + \overrightarrow{CB} + \overrightarrow{BA} + \overrightarrow{CA} + \overrightarrow{BC} + \overrightarrow{CA}$ \\
$\vec{u} = \overrightarrow{AB} + \overrightarrow{BA} + \overrightarrow{CA} + \overrightarrow{CA}$ \\
$\vec{u} = \vec{0} + 2\overrightarrow{CA}$

\vspace{1em}

\noindent \textbf{Résultat final :} $\vec{u} = 2\overrightarrow{CA}$

\noindent $\vec{u} = \overrightarrow{AA} + \overrightarrow{CA} + \overrightarrow{CA}$ \\
$\vec{u} = \vec{0} + 2\overrightarrow{CA}$

\vspace{1em}

\begin{center}
    \framebox{
        $\vec{u} = 2\overrightarrow{CA}$
    }
\end{center}

\section*{2- \underline{Milieu d'un segment}}

\subsection*{\underline{Propriété :}}
Soient $A$ et $B$ deux points distincts et $I$ milieu du segment $[AB]$. Alors, pour tout point $M$ du plan, on a :

\begin{center}
    \color{red}
    \framebox{\color{black}$\overrightarrow{MA} + \overrightarrow{MB} = 2\overrightarrow{MI}$}
    \quad \quad
    \framebox{\color{black}$\overrightarrow{MI} = \frac{1}{2} \left( \overrightarrow{MA} + \overrightarrow{MB} \right)$}
\end{center}

\vspace{1em}

\subsection*{\underline{Preuve :}}

\begin{center}
    \begin{tikzpicture}
        \draw[thick] (0,0) -- (4,0);
        \node at (0,0) [below] {$A$};
        \node at (0,0) {$\times$};
        \node at (2,0) [below] {$I$};
        \node at (2,0) {$\times$};
        \node at (4,0) [below] {$B$};
        \node at (4,0) {$\times$};
    \end{tikzpicture}
\end{center}

\noindent On a : $I$ milieu de $[AB]$ alors : \\
$\overrightarrow{AI} = \overrightarrow{IB}$ \\
$\overrightarrow{AM} + \overrightarrow{MI} = \overrightarrow{IM} + \overrightarrow{MB}$ \\
$-\overrightarrow{MA} + \overrightarrow{MI} = -\overrightarrow{MI} + \overrightarrow{MB}$ \\
$\overrightarrow{MI} + \overrightarrow{MI} = \overrightarrow{MB} + \overrightarrow{MA}$ \\
$2\overrightarrow{MI} = \overrightarrow{MB} + \overrightarrow{MA}$ \\
D'où $\overrightarrow{MA} + \overrightarrow{MB} = 2\overrightarrow{MI}$

\section*{\underline{\color{red}Application}}

\noindent Les questions \textbf{1)} et \textbf{2)} sont indépendantes.

\vspace{1em}

\noindent \textbf{1)} $ABCD$ est un quadrilatère quelconque avec $I$ et $J$ les milieux respectifs de $[AB]$ et $[CD]$. \\
Montrer que $\overrightarrow{AC} + \overrightarrow{BD} = 2\overrightarrow{IJ}$

\vspace{2em}

\noindent \textbf{2)} $ABC$ est un triangle quelconque avec $I$ milieu de $[BC]$. \\
Prouver que pour tout point du plan on a : \\
$2\overrightarrow{MA} - \overrightarrow{MB} - \overrightarrow{MC} = 2\overrightarrow{IA}$

\section*{\underline{\color{red}Solution}}

\noindent \textbf{1) Soient $I * [AB]$ et $J * [CD]$. Montrons que $\overrightarrow{AC} + \overrightarrow{BD} = 2\overrightarrow{IJ}$}

\vspace{1em}

\noindent On a :
\begin{align*}
\overrightarrow{AC} + \overrightarrow{BD} &= \overrightarrow{AC} + (\overrightarrow{BC} + \overrightarrow{CD}) \\
\overrightarrow{AC} + \overrightarrow{BD} &= (\overrightarrow{AC} + \overrightarrow{BC}) + \overrightarrow{CD}
\end{align*}

\noindent Or $I$ milieu de $[AB]$ donc $\overrightarrow{AC} + \overrightarrow{BC} = 2\overrightarrow{IC}$ \\
Et $J$ milieu de $[CD]$ donc $\overrightarrow{CD} = 2\overrightarrow{CJ}$ (Note: Correction selon le raisonnement du milieu) \\
\noindent Alors $\overrightarrow{AC} + \overrightarrow{BD} = 2\overrightarrow{IC} + 2\overrightarrow{CJ}$
\begin{align*}
\overrightarrow{AC} + \overrightarrow{BD} &= 2(\overrightarrow{IC} + \overrightarrow{CJ}) \\
\overrightarrow{AC} + \overrightarrow{BD} &= 2(\overrightarrow{IJ})
\end{align*}

\noindent \textbf{D'où $\overrightarrow{AC} + \overrightarrow{BD} = 2\overrightarrow{IJ}$}

\vspace{2em}

\noindent \textbf{2) Montrons que $2\overrightarrow{MA} - \overrightarrow{MB} - \overrightarrow{MC} = 2\overrightarrow{IA}$}

\vspace{1em}

\noindent On a : $2\overrightarrow{MA} - \overrightarrow{MB} - \overrightarrow{MC} = 2\overrightarrow{MA} + \overrightarrow{BM} + \overrightarrow{CM}$ \\
\noindent Or $I * [BC]$ donc $\overrightarrow{BM} + \overrightarrow{CM} = 2\overrightarrow{IM}$ \\
\noindent Alors $2\overrightarrow{MA} - \overrightarrow{MB} - \overrightarrow{MC} = 2\overrightarrow{MA} + 2\overrightarrow{IM}$
\begin{align*}
&= 2(\overrightarrow{MA} + \overrightarrow{IM}) \\
&= 2(\overrightarrow{IM} + \overrightarrow{MA}) \\
&= 2(\overrightarrow{IA})
\end{align*}

\noindent \textbf{$= 2\overrightarrow{IA}$}


\section*{3- \underline{La construction d'une somme de vecteurs}}

\subsection*{a- \underline{À l'aide de la relation de Chasles}}

\noindent Soient $\vec{u}$ et $\vec{v}$ deux vecteurs, $A$, $B$ et $C$ 3 points tels que $\vec{u} = \overrightarrow{AB}$ et $\vec{v} = \overrightarrow{BC}$. \\
La somme des vecteurs $\vec{u}$ et $\vec{v}$ est notée $\vec{u} + \vec{v}$ et est donnée par :

\begin{center}
    \color{red}
    \framebox{\color{black} $\vec{u} + \vec{v} = \overrightarrow{AC}$}
\end{center}

\vspace{1em}

\subsection*{\underline{Exemple :}}
\noindent Soient $\vec{u}$ et $\vec{v}$ deux vecteurs donnés. Construire le vecteur $\vec{w}$ tel que $\vec{w} = \vec{u} + \vec{v}$.

\vspace{1em}

\subsection*{\underline{Correction}}

\begin{center}
    \begin{tikzpicture}[>=stealth, thick, scale=1.2]
        % Construction principale (Triangle ABC)
        \coordinate (A) at (0,2);
        \coordinate (B) at (1,0.8);
        \coordinate (C) at (0.5,-0.5);
        
        \draw[->] (A) -- (B) node[midway, right] {$\vec{u}$};
        \draw[->] (B) -- (C) node[midway, right] {$\vec{v}$};
        \draw[->, blue] (A) -- (C) node[midway, left] {$\vec{w}$};
        
        \node[above] at (A) {$A$};
        \node[right] at (B) {$B$};
        \node[below] at (C) {$C$};

        % Vecteurs isolés (données de l'énoncé)
        \begin{scope}[shift={(2.5,1.5)}]

        \end{scope}
        
        \begin{scope}[shift={(3.5,0)}]
            \draw[->] (0,0) -- (0.5,-1.2) node[midway, right] {$\vec{u}$};
        \end{scope}
    \end{tikzpicture}
\end{center}

\subsection*{b- \underline{À l'aide de la règle du parallélogramme}}

\noindent Soient $\vec{u}$ et $\vec{v}$ deux vecteurs, $A$, $B$ et $C$ trois points non alignés et $D$ un point tels que : $\vec{u} = \overrightarrow{AB}$ et $\vec{v} = \overrightarrow{AC}$. \\
On a : \\
$\overrightarrow{AB} + \overrightarrow{AC} = \overrightarrow{AD}$ ssi $ABDC$ est un parallélogramme. \\
Ainsi, la somme des vecteurs $\vec{u}$ et $\vec{v}$ notée $\vec{u} + \vec{v}$ est donnée par :

\begin{center}
    \color{red}
    \framebox{\color{black} $\vec{u} + \vec{v} = \overrightarrow{AD}$}
\end{center}

\vspace{1em}

\begin{center}
    \begin{tikzpicture}[>=stealth, thick, scale=1.5]
        % Points du parallélogramme
        \coordinate (A) at (0,0);
        \coordinate (B) at (1,0.8);
        \coordinate (C) at (1.2,-0.5);
        \coordinate (D) at ($(B)+(C)-(A)$); % Calcul automatique du point D

        % Vecteurs de base
        \draw[->] (A) -- (B) node[midway, above left] {$\vec{u}$};
        \draw[->] (A) -- (C) node[midway, below left] {$\vec{v}$};
        
        % Somme (Diagonale)
        \draw[->, very thick] (A) -- (D) node[midway, above] {$\vec{u} + \vec{v}$};

        % Côtés en pointillés
        \draw[dashed] (B) -- (D);
        \draw[dashed] (C) -- (D);

        % Étiquettes des points
        \node[left] at (A) {$A$};
        \node[above] at (B) {$B$};
        \node[below] at (C) {$C$};
        \node[right] at (D) {$D$};
    \end{tikzpicture}
\end{center}

\noindent \underline{\color{red}Ex :} Soient $\vec{u}$ et $\vec{v}$ deux vecteurs donnés. Construire le vecteur $\vec{w}$ tel que $\vec{w} = \vec{u} + \vec{v}$

\vspace{1.5em}

\begin{center}
    \underline{\color{red}Correction}
\end{center}

\noindent Soient $A$, $B$ et $C$ 3 points non alignés tels que : \\
$\vec{u} = \overrightarrow{AB}$, $\vec{v} = \overrightarrow{AC}$ et $\overrightarrow{AB} + \overrightarrow{AC} = \overrightarrow{AD}$. Alors $\vec{w} = \overrightarrow{AD}$

\vspace{2em}

\begin{center}
    \begin{tikzpicture}[>=stealth, thick, scale=1.3]
        % Vecteurs de données (témoins)
        \draw[->] (-1,2) -- (0.5,3.5) node[midway, above left] {$\vec{u}$};
        \draw[->] (2.5,2.5) -- (3.8,1.2) node[midway, above right] {$\vec{v}$};

        % Construction du parallélogramme
        \coordinate (A) at (0,0);

\section*{4- \underline{Multiplication d'un vecteur par un nombre réel}}

\subsection*{\underline{Activité}}

\noindent Soient $I$ le milieu d'un segment $[AB]$ et $J$ celui du segment $[IB]$.

\vspace{1em}

\begin{enumerate}
    \item[i)] Comparer les caractéristiques des vecteurs $\overrightarrow{AB}$ et $\overrightarrow{AI}$.
    \item[ii)] Comparer les caractéristiques des vecteurs $\overrightarrow{IB}$ et $\overrightarrow{BA}$.
\end{enumerate}

\section*{\underline{\color{red}Correction}}

\begin{center}
    \begin{tikzpicture}[thick]
        % Ligne du segment
        \draw (0,0) -- (6,0);
        
        % Graduations et points
        \draw (0,0.1) -- (0,-0.1) node[below] {$A$};
        \draw (3,0.1) -- (3,-0.1) node[below] {$I$};
        \draw (4.5,0.1) -- (4.5,-0.1) node[below] {$J$};
        \draw (6,0.1) -- (6,-0.1) node[below] {$B$};
    \end{tikzpicture}
\end{center}

\vspace{1em}

\noindent \textbf{i) Comparons les caractéristiques des vecteurs $\overrightarrow{AB}$ et $\overrightarrow{AI}$} \\
$\overrightarrow{AB}$ et $\overrightarrow{AI}$ ont \textbf{même direction}, \textbf{même sens} tel que $AB = 2AI$.

\vspace{0.5em}

\noindent \textbf{ii) Comparons les caractéristiques des vecteurs $\overrightarrow{JB}$ et $\overrightarrow{BA}$} \\
$\overrightarrow{JB}$ et $\overrightarrow{BA}$ ont \textbf{même direction}, de \textbf{sens opposés} tel que : \\
$BA = 4JB$.

\subsection*{\underline{\color{red}Définition :}}

\noindent Soient $\vec{u}$ un vecteur et $k$ un nombre réel ($k \in \mathbb{R}$). \\
On appelle \textbf{produit du vecteur $\vec{u}$ par le réel $k$}, le vecteur noté $k\vec{u}$ qui possède les caractéristiques suivantes :

\begin{itemize}
    \item Sa \textbf{direction} est celle de $\vec{u}$ (si $\vec{u} \neq \vec{0}$ et $k \neq 0$).
    \item Sa \textbf{norme} est $\|k\vec{u}\| = |k| \times \|\vec{u}\|$.
    \item Son \textbf{sens} dépend du signe de $k$ :
    \begin{itemize}
        \item[\textcolor{red}{$\bullet$}] Si $k > 0$, alors $k\vec{u}$ et $\vec{u}$ sont de \textbf{même sens}.
        \item[\textcolor{red}{$\bullet$}] Si $k < 0$, alors $k\vec{u}$ et $\vec{u}$ sont de \textbf{sens opposés}.
    \end{itemize}
\end{itemize}

\subsection*{\underline{\color{red}Exemple :}}

\noindent \textbf{i)} $\vec{v} = 2 \vec{u}$ \\
$\vec{v}$ et $\vec{u}$ ont \textbf{même direction} et \textbf{même sens} tel que : \\
$\|\vec{v}\| = |2| \|\vec{u}\| = 2 \|\vec{u}\|$

\vspace{1.5em}

\noindent \textbf{ii)} $\vec{v} = -4 \vec{u}$ \\
$\vec{v}$ et $\vec{u}$ ont \textbf{même direction} mais de \textbf{sens opposés} tel que : \\
$\|\vec{v}\| = |-4| \|\vec{u}\|$ \\
$\|\vec{v}\| = 4 \|\vec{u}\|$

\subsection*{\underline{\color{red}Remarque :}}

\noindent Pour tout vecteur $\vec{u}$ et pour tout réel $\lambda$, on a :

\begin{enumerate}
    \item[i)] $0 \cdot \vec{u} = \vec{0}$
    \item[ii)] $\lambda \cdot \vec{0} = \vec{0}$
\end{enumerate}
\section*{5- \underline{Combinaison linéaire de vecteurs}}

\subsection*{a- \underline{Combinaison linéaire de deux vecteurs}}

\subsubsection*{\underline{\color{red}Définition}}

\noindent Soient $\vec{u}$ et $\vec{v}$ deux vecteurs, $a$ et $b$ des réels. On appelle \textbf{combinaison linéaire de $\vec{u}$ et $\vec{v}$ de coefficients respectifs $a$ et $b$}, le vecteur $a\vec{u} + b\vec{v}$.

\vspace{1em}

\noindent \underline{\textbf{Ex :}} \\
$\vec{w} = 2\vec{u} - 5\vec{v}$ \\
Alors $\vec{w}$ est une combinaison linéaire des vecteurs $\vec{u}$ et $\vec{v}$ de coefficients respectifs $2$ et $-5$.

\subsection*{b- \underline{Vecteurs colinéaires}}

\subsubsection*{\underline{\color{red}Définition}}

\noindent Deux vecteurs $\vec{u}$ et $\vec{v}$ sont \textbf{colinéaires} ssi il existe un réel $K$ tel que $\vec{u} = K\vec{v}$ et $\vec{v} = K'\vec{u}$.

\vspace{1em}

\subsection*{\underline{\color{red}Exemple :}}

\noindent \textbf{i)} $\vec{u} = -2\vec{v} \quad (K = -2)$ \\
$\vec{u}$ et $\vec{v}$ sont \textbf{colinéaires}.

\vspace{1em}

\noindent \textbf{ii)} $\vec{v} = 5\vec{u} \quad (K = 5)$ \\
$\vec{u}$ et $\vec{v}$ sont \textbf{colinéaires}.

\vspace{1.5em}

\subsection*{\underline{\color{red}Remarque :}}

\noindent Le vecteur nul est \textbf{colinéaire} à tous les vecteurs. \\
En effet, pour tout vecteur $\vec{u}$, on a : \\
$\vec{0} = 0\vec{u} \quad (K = 0)$

\noindent \underline{NB :} \\
En général pour montrer que deux droites sont parallèles ou bien que 3 points sont alignés, on utilise la notion de colinéarité.

\vspace{1.5em}

\section*{\underline{\color{red}Application}}

\noindent $ABC$ un triangle quelconque. \\
\textbf{1)} Construire $M$ et $N$ tels que $\overrightarrow{AM} = -\frac{2}{3} \overrightarrow{AB}$ et $\overrightarrow{AN} = -\frac{2}{3} \overrightarrow{AC}$

\vspace{1em}

\noindent \textbf{2)} Démontrer que $(MN)$ et $(BC)$ sont parallèles.

\vspace{1em}

\noindent \textbf{3)} Soient $S'$ et $T$ les milieux respectifs de $[BC]$ et $[MN]$. \\
Démontrer que $A$, $S'$ et $T$ sont alignés.

\section*{\underline{\color{red}Solution de l'Application}}
\begin{center}
    \begin{tikzpicture}[scale=1.5, >=stealth, thick]
        % Définition des points du triangle ABC
        \coordinate (A) at (0,0);
        \coordinate (B) at (3,-1);
        \coordinate (C) at (2,2);
        
        % Calcul des points M et N (AM = -2/3 AB et AN = -2/3 AC)
        \coordinate (M) at ($ (A) ! -0.666 ! (B) $);
        \coordinate (N) at ($ (A) ! -0.666 ! (C) $);
        
        % Calcul des milieux S' (de BC) et T (de MN)
        \coordinate (S) at ($ (B) ! 0.5 ! (C) $);
        \coordinate (T) at ($ (M) ! 0.5 ! (N) $);
        
        % Tracé du triangle et des segments
        \draw (A) -- (B) -- (C) -- cycle;
        \draw[red] (M) -- (N);
        \draw[blue, dashed] (S) -- (T); % Ligne d'alignement passant par A
        
        % Prolongements pour montrer M et N
        \draw[dashed] (A) -- (M);
        \draw[dashed] (A) -- (N);
        
        % Marquage des points
        \node[below left] at (A) {$A$};
        \node[below right] at (B) {$B$};
        \node[above right] at (C) {$C$};
        \node[above right] at (M) {$M$};
        \node[below left] at (N) {$N$};
        \node[right] at (S) {$S'$};
        \node[left] at (T) {$T$};
        
        % Optionnel : ajout de petites croix pour les milieux
        \node at (S) {$\times$};
        \node at (T) {$\times$};
    \end{tikzpicture}
\end{center}

\subsection*{1) Construction (Description)}
\noindent Pour construire $M$ et $N$, on divise les segments $[AB]$ et $[AC]$ en trois parties égales. Comme le coefficient est $-\frac{2}{3}$, les points $M$ et $N$ sont placés sur les demi-droites opposées à $[AB)$ et $[AC)$ en partant de $A$.

\vspace{1em}

\subsection*{2) Montrons que $(MN) // (BC)$}
\noindent Exprimons le vecteur $\overrightarrow{MN}$ en fonction de $\overrightarrow{BC}$ en utilisant la relation de Chasles :
\begin{align*}
\overrightarrow{MN} &= \overrightarrow{MA} + \overrightarrow{AN} \\
\overrightarrow{MN} &= -\overrightarrow{AM} + \overrightarrow{AN}
\end{align*}
\noindent En remplaçant par les données de l'énoncé :
\begin{align*}
\overrightarrow{MN} &= -\left( -\frac{2}{3}\overrightarrow{AB} \right) + \left( -\frac{2}{3}\overrightarrow{AC} \right) \\
\overrightarrow{MN} &= \frac{2}{3}\overrightarrow{AB} - \frac{2}{3}\overrightarrow{AC} \\
\overrightarrow{MN} &= -\frac{2}{3}(\overrightarrow{AC} - \overrightarrow{AB}) \\
\overrightarrow{MN} &= -\frac{2}{3}\overrightarrow{BC}
\end{align*}
\noindent Comme $\overrightarrow{MN} = k\overrightarrow{BC}$ (avec $k = -\frac{2}{3}$), les vecteurs $\overrightarrow{MN}$ et $\overrightarrow{BC}$ sont \textbf{colinéaires}. \\
\textbf{Conclusion :} Les droites $(MN)$ et $(BC)$ sont parallèles.

\vspace{1em}

\subsection*{3) Montrons que $A$, $S'$ et $T$ sont alignés}
\noindent On sait que $S'$ est le milieu de $[BC]$ et $T$ est le milieu de $[MN]$. \\
D'après la propriété du milieu, on a :
\[ \overrightarrow{AS'} = \frac{1}{2}(\overrightarrow{AB} + \overrightarrow{AC}) \quad \text{et} \quad \overrightarrow{AT} = \frac{1}{2}(\overrightarrow{AM} + \overrightarrow{AN}) \]
\noindent Remplaçons $\overrightarrow{AM}$ et $\overrightarrow{AN}$ :
\begin{align*}
\overrightarrow{AT} &= \frac{1}{2}\left( -\frac{2}{3}\overrightarrow{AB} - \frac{2}{3}\overrightarrow{AC} \right) \\
\overrightarrow{AT} &= \frac{1}{2} \times \left( -\frac{2}{3} \right) (\overrightarrow{AB} + \overrightarrow{AC}) \\
\overrightarrow{AT} &= -\frac{2}{3} \left[ \frac{1}{2}(\overrightarrow{AB} + \overrightarrow{AC}) \right] \\
\overrightarrow{AT} &= -\frac{2}{3} \overrightarrow{AS'}
\end{align*}
\noindent Le vecteur $\overrightarrow{AT}$ est un multiple du vecteur $\overrightarrow{AS'}$, ils sont donc \textbf{colinéaires}. Ayant le point $A$ en commun, les points $A$, $S'$ et $T$ sont \textbf{alignés}.
