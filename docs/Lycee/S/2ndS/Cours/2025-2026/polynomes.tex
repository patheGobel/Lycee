\subsection*{%
    \color{red} \underline{1 - Définitions}%
}

% --- DÉFINITION ---
\textbf{\underline{\textcolor{red}{Définition 1 :}}}

On appelle \textbf{\textcolor{red}{monôme}} à variable $x$ toute expression qui s'écrit sous la forme $a_n x^n$ où $n \in \mathbb{N}$, $a_n \in \mathbb{R}$.

\begin{itemize}
    \item $a_n$ est appelé le \textbf{\textcolor{red}{coefficient}} du monôme.
    \item $n$ est appelé \textbf{\textcolor{red}{degré}} du monôme si $a_n \neq 0$.
\end{itemize}

\textbf{\underline{\textcolor{red}{Exemples :}}}
\begin{enumerate}
    \item $2x^8 : n = 8$ et $a_8 = 2$ \\
    $2x^8$ est un monôme de degré $8$ et de coefficient $2$.
    
    \item $-5x^4 : n = 4$ et $a_4 = -5$ \\
    $-5x^4$ est un monôme de degré $4$ et de coefficient $-5$.
\end{enumerate}

\textbf{\underline{\textcolor{red}{Remarque :}}} \\
Toute constante $C \neq 0$ est un monôme de degré $0$. En effet, si $C$ est une constante, alors $C = C \cdot x^0$ ($x \neq 0$). \\
\textbf{\underline{\textcolor{red}{Exemple :}}} $5, 4, 8$ sont des monômes de degré $0$.

\vspace{0.5cm}

\textbf{\underline{\textcolor{red}{Définition 2 :}}} 

Soit $n \in \mathbb{N}$ \\
On appelle \textbf{\textcolor{red}{polynôme}} (somme de plusieurs monômes) de degré $n$, toute expression qui s'écrit sous la forme :
\[ \textcolor{red}{a_n x^n + a_{n-1} x^{n-1} + \dots + a_1 x + a_0} \]
où \textcolor{red}{$a_n, a_{n-1}, \dots, a_1, a_0$} sont des réels appelés \textbf{\textcolor{red}{coefficients}} du polynôme avec $a_n \neq 0$.

\textbf{\underline{\textcolor{red}{Exemples :}}}

\begin{enumerate}
    \item $P(x) = 3x^2 + 4x + 8$ : \\
    $n = 2$, $a_2 = 3$, $a_1 = 4$, $a_0 = 8$ \\
    $P$ est un polynôme de degré $2$.
    
    \item $Q(x) = -5x^6 + 4x^4 + x + 4$ \\
    $n = 6$, $a_6 = -5$, $a_5 = 0$, $a_4 = 4$, $a_3 = 0$, $a_2 = 0$, $a_1 = 1$, $a_0 = 4$ \\
    $Q$ est un polynôme de degré $6$.
\end{enumerate}

\vspace{0.5cm}

\textbf{\underline{\textcolor{red}{Définition 3 :}}} 

Un polynôme est dit \textbf{\textcolor{red}{nul}} lorsque tous ses coefficients sont nuls.

\textbf{\underline{\textcolor{red}{Notation :}}}

\begin{itemize}
    \item[i)] Si $P$ est un polynôme, on note son degré par \textcolor{red}{$d^{\circ}p$} ou \textcolor{red}{$\deg(p)$}.
    \item[ii)] Le polynôme nul est noté $0$ tel que : \\
    \textcolor{red}{$\deg(0) = -\infty$} (Par convention).
\end{itemize}

% --- SUBSECTION 1 ---
\subsection*{%
    \color{red} \underline{2 - Égalités de deux polynômes}%
}

\textbf{\underline{\textcolor{red}{Théorème :}}} Deux polynômes $P$ et $Q$ sont \textcolor{red}{égaux} ssi :
\begin{itemize}
    \item[i)] $\deg(P) = \deg(Q)$
    \item[ii)] Les coefficients des monômes de même degré de $P$ et $Q$ sont \textcolor{red}{égaux}.
\end{itemize}

\textbf{\underline{\textcolor{red}{Preuve :}}} T.P.E (Travail Personnel de l'Élève)

\vspace{0.5cm}

\begin{center}
    \textbf{\underline{\textcolor{red}{Application}}}
\end{center}

Soit $P(x) = 2x^3 - 5x^2 + x + 2$ \\
et $Q(x) = (x - 2)(ax^2 + bx + c)$

\begin{enumerate}
    \item Développer, réduire et ordonner $Q(x)$.
    \item Déterminer les réels $a, b$ et $c$ pour que l'on ait $P(x) = Q(x)$ pour tout réel $x$.
\end{enumerate}

\begin{center}
    \textbf{\underline{\textcolor{red}{Correction}}}
\end{center}

\textbf{1 - Développons, réduisons et ordonnons $Q(x)$.}
$Q(x) = (x - 2)(ax^2 + bx + c)$ \\
$Q(x) = ax^3 + bx^2 + cx - 2ax^2 - 2bx - 2c$ \\
$Q(x) = ax^3 + bx^2 - 2ax^2 + cx - 2bx - 2c$ \\
\fbox{$Q(x) = ax^3 + (b - 2a)x^2 + (c - 2b)x - 2c$}

\textbf{2 - Déterminons $a, b$ et $c$ tels que : $P(x) = Q(x)$}
On a : \\
$P(x) = 2x^3 - 5x^2 + x + 2$ \\
$Q(x) = ax^3 + (b - 2a)x^2 + (c - 2b)x - 2c$

\vspace{0.3cm}
Par \textbf{identification des coefficients}, on a :
\begin{align}
    a &= 2 \quad (1) & c - 2b &= 1 \quad (3) \\
    b - 2a &= -5 \quad (2) & -2c &= 2 \quad (4)
\end{align}

\fbox{\textbf{\textcolor{red}{$a = 2$}}}

\vspace{0.2cm}
$b - 2a = -5 \implies b = -5 + 2a$ \\
$b = -5 + 2(2)$ \\
$b = -5 + 4$ \\
\fbox{\textbf{\textcolor{red}{$b = -1$}}}

\vspace{0.2cm}
$c - 2b = 1 \implies c = 1 + 2b$ \\
$c = 1 + 2(-1)$ \\
$c = 1 - 2$ \\
\fbox{\textbf{\textcolor{red}{$c = -1$}}}

% --- SUBSECTION 3 ---
\subsection*{%
    \color{red} \underline{3 - Opérations sur les polynômes}%
}

% --- SUBSUBSECTION 1 ---
\subsubsection*{%
    \color{red} \underline{a - Somme de deux polynômes}%
}

\underline{Activité}
Soit $P(x) = -2x^4 + 3x^3 + x^2 - 4$ \\
$Q(x) = 2x^4 + x^3 - 2x + 1$

\begin{enumerate}
    \item Calculer $P(x) + Q(x)$. En déduire $\deg(P+Q)$.
    \item Comparer $\deg(P+Q)$ et $\max(\deg(P); \deg(Q))$.
\end{enumerate}

\underline{\textcolor{red}{Correction}}
\begin{enumerate}
    \item Calculons $P(x) + Q(x)$ : \\
    $P(x) + Q(x) = -2x^4 + 3x^3 + x^2 - 4 + 2x^4 + x^3 - 2x + 1$ \\
    $P(x) + Q(x) = -2x^4 + 2x^4 + 3x^3 + x^3 + x^2 - 2x - 4 + 1$ \\
    $P(x) + Q(x) = 0 + 4x^3 + x^2 - 2x - 3$ \\
    \fbox{$P(x) + Q(x) = 4x^3 + x^2 - 2x - 3$} \\
    Donc \textbf{$\deg(P+Q) = 3$}.

    \item On a :
    \[
    \begin{cases}
    \deg(P+Q) = 3 \\
    \max(\deg(P), \deg(Q)) = 4
    \end{cases}
    \]
    Donc \textbf{$\deg(P+Q) < \max(\deg(P); \deg(Q))$}.
\end{enumerate}

\textbf{\underline{\textcolor{red}{Théorème :}}} La somme de polynômes $P$ et $Q$ est un polynôme noté $P+Q$ tel que :
\[ \textcolor{red}{\deg(P+Q) \leq \max(\deg(P); \deg(Q))} \]

\vspace{0.5cm}

% --- SUBSUBSECTION 1 ---
\subsubsection*{%
    \color{red} \underline{b - Produit de deux polynômes}%
}

\textbf{\underline{\textcolor{red}{Activité}}}

Soit $P(x) = 2x^3 - x^2 + x + 1$ \\
$Q(x) = x^2 + x + 3$

\begin{enumerate}
    \item Calculer $P(x) \times Q(x)$. En déduire $\deg(P \times Q)$.
    \item Comparer $\deg(P \times Q)$ et $\deg(P) + \deg(Q)$.
\end{enumerate}

\textbf{\underline{\textcolor{red}{Correction}}}
\begin{enumerate}
    \item Calculons $P(x) \times Q(x)$ : \\
    $P(x) \times Q(x) = (2x^3 - x^2 + x + 1)(x^2 + x + 3)$ \\
    $P(x) \times Q(x) = 2x^5 + 2x^4 + 6x^3 - x^4 - x^3 - 3x^2 + x^3 + x^2 + 3x + x^2 + x + 3$ \\
    $P(x) \times Q(x) = 2x^5 + x^4 + 6x^3 - x^2 + 4x + 3$ \\
    \textbf{$\deg(P \times Q) = 5$}.

    \item Comparons $\deg(P \times Q)$ et $\deg(P) + \deg(Q)$ : \\
    $\deg(P \times Q) = 5$ \\
    $\left. \begin{array}{l} \deg(P) = 3 \\ \deg(Q) = 2 \end{array} \right\} \implies \deg(P) + \deg(Q) = 5$ \\
    Donc \textbf{$\deg(P \times Q) = \deg(P) + \deg(Q)$}.
\end{enumerate}

\textbf{\underline{\textcolor{red}{Théorème :}}} Le produit de deux polynômes $P$ et $Q$ est un polynôme noté $P \times Q$ tel que :
\[ \textcolor{red}{\deg(P \times Q) = \deg(P) + \deg(Q)} \]

\subsection*{%
    \color{red} \underline{4 - Factorisation d'un polynôme}%
}

\subsubsection*{%
    \color{red} \underline{a - Définition}%
}

Un réel $\alpha$ est \textbf{\textcolor{red}{racine}} (ou \textbf{\textcolor{red}{zéro}}) d'un polynôme $P$ si \fbox{\textcolor{red}{$P(\alpha) = 0$}}.

\textbf{\underline{\textcolor{red}{Exemples :}}} Soit $P(x) = 3x^3 - 2x^2 + 4x - 5$
\begin{enumerate}
    \item On a : $P(1) = 3(1)^3 - 2(1)^2 + 4(1) - 5 = 3 - 2 + 4 - 5 = 0$. \\
    Donc \textbf{1 est racine de $P$}.
    \item On a : $P(-1) = 3(-1)^3 - 2(-1)^2 + 4(-1) - 5 = -3 - 2 - 4 - 5 = -14$. \\
    Donc \textbf{$-1$ n'est pas racine de $P$} car $P(-1) \neq 0$.
\end{enumerate}

\subsubsection*{%
    \color{red} \underline{b - Théorème}%
}

Soient $P$ un polynôme et $\alpha$ un réel : \\
\textcolor{red}{$\alpha$ est racine de $P$ ssi $P(x) = (x - \alpha) Q(x)$} \\
Où $Q$ est un polynôme tel que : \textcolor{red}{$\deg(Q) = \deg(P) - 1$}.

\begin{center}
    \textbf{\underline{\textcolor{red}{Application}}}
\end{center}

Soit $P(x) = 2x^3 - 5x^2 - x + 6$
\begin{enumerate}
    \item Montrer que 2 est racine de $P$.
    \item En déduire une factorisation de $P(x)$.
\end{enumerate}

\begin{center}
    \textbf{\underline{\textcolor{red}{Correction}}}
\end{center}

1) Montrons que 2 est racine de $P$ : \\
$P(2) = 2(2)^3 - 5(2)^2 - 2 + 6 = 2(8) - 5(4) - 2 + 6 = 16 - 20 - 2 + 6 = 0$. \\
Donc \textbf{2 est racine de $P$}.

2) Déduisons-en une factorisation de $P(x)$ : \\
2 racine de $P \implies P(x) = (x - 2) Q(x)$ avec $\deg(Q) = 3 - 1 = 2$. \\
Ainsi, pour déterminer $Q(x)$, on distingue \textbf{3 méthodes} :

\paragraph{\underline{Méthode 1 : Identification des coefficients}}
On a $\deg(Q) = 2$ alors $Q(x) = ax^2 + bx + c$ où $a, b, c \in \mathbb{R}$ avec $a \neq 0$. \\
$P(x) = (x - 2)(ax^2 + bx + c) = ax^3 + bx^2 + cx - 2ax^2 - 2bx - 2c$ \\
$P(x) = ax^3 + (b - 2a)x^2 + (c - 2b)x - 2c$ \\
Or $P(x) = 2x^3 - 5x^2 - x + 6$. Par identification :
\[
\begin{cases}
a = 2 \\
b - 2a = -5 \implies b = -5 + 2(2) = -1 \\
c - 2b = -1 \implies c = -1 + 2(-1) = -3 \\
-2c = 6 \implies c = -3
\end{cases}
\]
\fbox{$Q(x) = 2x^2 - x - 3$} \quad D'où \fbox{$P(x) = (x - 2)(2x^2 - x - 3)$}

\paragraph{\underline{Méthode 2 : Méthode d'Horner}}
\begin{center}
\begin{tabular}{|c|c|c|c|c|}
\hline
Coef $P$ & 2 & -5 & -1 & 6 \\ \hline
2 & $\downarrow$ & 4 & -2 & -6 \\ \hline
Coef $Q$ & 2 & -1 & -3 & \textbf{0} \\ \hline
\end{tabular}
\end{center}
$Q(x) = 2x^2 - x - 3$ d'où $P(x) = (x - 2)(2x^2 - x - 3)$.


\paragraph{\underline{Méthode 3 : Méthode de la division euclidienne}}
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{r|l}
$2x^3 - 5x^2 - x + 6$ & $x - 2$ \\ \cline{2-2}
$- (2x^3 - 4x^2)$ & $2x^2 - x - 3$ \\ \cline{1-1}
$-x^2 - x + 6$ & \\
$- (-x^2 + 2x)$ & \\ \cline{1-1}
$-3x + 6$ & \\
$- (-3x + 6)$ & \\ \cline{1-1}
$0$ & 
\end{tabular}
\end{center}

\paragraph{\underline{\textcolor{red}{Exercice : T.P.E}}}

Soit $P(x) = 3x^3 + x + 4$

\begin{enumerate}
    \item Montrer que $-1$ est racine de $P$
    \item En déduire une factorisation de $P(x)$ (utiliser les 3 méthodes).
\end{enumerate}

\vspace{1cm}

\noindent \textbf{\underline{\textcolor{red}{Remarque :}}} \\

Soit $P$ un polynôme de degré $n$. \\
Les réels $\alpha_1, \alpha_2, \dots, \alpha_m$ ($1 \le m \le n$) sont racines de $P$ si :
\[ P(x) = (x - \alpha_1)(x - \alpha_2) \times \dots \times (x - \alpha_m) Q(x) \]
Où $Q$ est un polynôme tel que $d^\circ Q = n - m$.

\begin{center}
    \textbf{\underline{\textcolor{red}{Application}}}
\end{center}
Soit $P(x) = 3x^4 + x^3 - 11x^2 - 4x - 4$

\begin{enumerate}
    \item Montrer que $2$ et $-2$ sont racines de $P(x)$
    \item En déduire $P(x) = (x - 2)(x + 2) Q(x)$ où $Q$ est un polynôme à déterminer.
\end{enumerate}

\begin{center}
    \textbf{\underline{\textcolor{red}{Correction}}}
\end{center}

\noindent \textbf{1) Montrons que $2$ et $-2$ sont racines de $P(x)$} \\
On a : \\
$P(2) = 3(2)^4 + (2)^3 - 11(2)^2 - 4(2) - 4$ \\
$P(2) = 0$ \\
Donc $2$ est racine de $P$.

\vspace{0.5em}

\noindent On a : \\
$P(-2) = 3(-2)^4 + (-2)^3 - 11(-2)^2 - 4(-2) - 4$ \\
$P(-2) = 0$ \\
Donc $-2$ est racine de $P$.

\vspace{1em}

\noindent \textbf{2) Déduisons-en que $P(x) = (x-2)(x+2)Q(x)$} \\
$2$ et $-2$ étant racines de $P$, alors on a : \\
$P(x) = (x-2)(x-(-2))Q(x)$ \\
D'où $P(x) = (x-2)(x+2)Q(x)$ \\
Où $Q$ est un polynôme tel que $\text{deg}(Q) = 4 - 2 = 2$.

\vspace{1em}

\noindent \textbf{Déterminons $Q(x)$} \\
% Tableau de Horner
\begin{center}
\renewcommand{\arraystretch}{1.5}
\begin{tabular}{|c|c|c|c|c|c|c|}
\hline
& & 3 & 1 & -11 & -4 & -4 \\ \hline
2 & & $\downarrow$ & 6 & 14 & 6 & 4 \\ \hline
& & 3 & 7 & 3 & 2 & \textbf{0} \\ \hline
-2 & & $\downarrow$ & -6 & -2 & -2 & \multicolumn{1}{c}{} \\ \cline{1-5}
& & 3 & 1 & 1 & \textbf{0} & \multicolumn{1}{c}{} \\ \cline{1-5}
\end{tabular}
\end{center}

\vspace{1em}

\noindent Donc \fbox{$Q(x) = 3x^2 + x + 1$}

\vspace{1em}

\noindent D'où \fbox{$P(x) = (x+2)(x-2)(3x^2 + x + 1)$}

\section*{%
    \color{red}II - \underline{ Fraction rationnelle}%
}

% --- SUBSECTION 1 ---
\subsection*{%
    \color{red} \underline{1 - Définitions}%
}
On appelle fraction rationnelle le rapport $\dfrac{P}{q}$ où $P$ et $q$ sont des polynômes.

\paragraph{Exemple :}
\[ f(x) = \frac{3x^2 + 4x + 1}{x - 4} \]
$f$ est une fraction rationnelle.

% --- SUBSECTION 1 ---
\subsection*{%
    \color{red} \underline{2 - Domaine de définition d'une fraction rationnelle}%
}
Soit $f = \dfrac{P}{q}$ où $P$ et $q$ sont des polynômes.

\vspace{1em}

$f(x)$ existe ssi $q(x) \neq 0$.

L'ensemble noté $D_f$ défini par :
\[ D_f = \{x \in \mathbb{R} \mid q(x) \neq 0\} \]
est appelé \textbf{domaine de définition de $f$}.

\begin{center}
    \textbf{\underline{\textcolor{red}{Application}}}
\end{center}

\begin{enumerate}
    \item Soit $P(x) = 3x^3 + x^2 + 3x + 5$
    \begin{enumerate}
        \item Montrer que $-1$ est racine de $P(x)$.
        \item En déduire une factorisation de $P(x)$.
    \end{enumerate}
    
    \item Soit $f(x) = \dfrac{3x^3 + x^2 + 3x + 5}{x^2 + 3x + 2}$
    \begin{enumerate}
        \item Déterminer $D_f$.
        \item Simplifier l'expression de $f(x)$ pour tout réel $x \in D_f$.
        \item Résoudre $f(x) \le 0$.
    \end{enumerate}
\end{enumerate}

\begin{center}
    \textbf{\underline{\textcolor{red}{Correction}}}
\end{center}

\textbf{\underline{\textcolor{red}{Exemples :}}}
\begin{enumerate}
    \item $x^2 - 1 = (x - 1)(x + 1)$ \\
    Donc $q(x) = x - 1$ divise $P(x) = x^2 - 1$.
    \item $x^3 - 8 = x^3 - 2^3$ \\
    $x^3 - 8 = (x - 2)(x^2 + 2x + 4)$ \\
    Donc $q(x) = x - 2$ divise $P(x) = x^3 - 8$.
\end{enumerate}

\noindent \underline{Propriété :} Soient $P$ et $q$ deux polynômes tels que $q$ divise $P$ ($q \mid P$). Alors toute racine de $q$ est racine de $P$.

\vspace{1cm}

\begin{center}
    \textbf{\underline{\textcolor{red}{Application}}}
\end{center}

\noindent Déterminer les réels $a$ et $b$ de telle sorte que le polynôme $x^4 + ax^2 + b$ soit divisible par $x^2 - 6x + 5$.

\begin{center}
    \textbf{\underline{\textcolor{red}{Correction}}}
\end{center}

Posons $P(x) = x^4 + ax^2 + b$ \\
et $q(x) = x^2 - 6x + 5$

$q \mid P \implies$ toute racine de $q$ est racine de $P$.

Déterminons les racines de $q$ : \\
$q(x) = x^2 - 6x + 5$ \\
$\Delta = (-6)^2 - 4(1)(5)$ \\
$\Delta = 36 - 20$ \\
$\Delta = 16 \implies \sqrt{\Delta} = 4$

$x_1 = \dfrac{6 - 4}{2} = 1 \quad \text{et} \quad x_2 = \dfrac{6 + 4}{2} = 5$

$1$ et $5$ sont racines de $q$, donc $1$ et $5$ sont racines de $P$.

Alors on a : 
$\begin{cases} P(1) = 0 \\ P(5) = 0 \end{cases}$

$P(1) = (1)^4 + a(1)^2 + b = 1 + a + b$

$P(5) = (5)^4 + a(5)^2 + b = 625 + 25a + b$

Ainsi, on a :
$\begin{cases} 1 + a + b = 0 \\ 625 + 25a + b = 0 \end{cases} \implies -1 \times \begin{cases} a + b = -1 & (E_1) \\ 25a + b = -625 & (E_2) \end{cases}$

$\begin{cases} -a - b = 1 \\ 25a + b = -625 \end{cases}$

\rule{2cm}{0.4pt} \\
$24a + 0 = -624$

$a = \dfrac{-624}{24}$

\fbox{$a = -26$}

Remplaçons la valeur de $a$ dans $(E_1)$ :
$-26 + b = -1$
$b = -1 + 26$

\fbox{$b = 25$}

% --- SECTION III ---
\section*{%
    \color{red}III - \underline{ Division euclidienne de deux polynômes}%
}


\subsection*{%
    \color{red}1 - \underline{ Divisibilité}%
}

\textbf{\underline{\textcolor{red}{Définition :}}}  

Soient $p$ et $q$ deux polynômes tels que $q \neq 0$.

\vspace{1em}

\noindent On dit que $q$ \textbf{\color{red} divise} $p$ et on note $q | p$, s'il existe un polynôme $Q$ tel que :
$$P = q \cdot Q$$

\noindent $P$ est appelé \textbf{\color{red} multiple} de $q$. En d'autres termes, on dit que $P$ est \textbf{\color{red} divisible} par $q$.

% --- Page 1 : Exemples ---
\textbf{\underline{\textcolor{red}{Exemples :}}} \\
1) $x^2 - 1 = (x - 1)(x + 1)$ \\
Donc $q(x) = x - 1$ divise $P(x) = x^2 - 1$ \\
2) $x^3 - 8 = x^3 - 2^3$ \\
$x^3 - 8 = (x - 2)(x^2 + 2x + 4)$ \\
Donc $q(x) = x - 2$ divise $P(x) = x^3 - 8$

\vspace{1.5em}

% --- Page 2 : Propriété et Application ---
\textbf{\underline{\textcolor{red}{Propriété :}}}  

Soient $P$ et $q$ deux polynômes tels que $q$ divise $P$ ($q | P$). Alors toute racine de $q$ est racine de $P$.

\vspace{1em}

\begin{center}
    \textbf{\underline{\textcolor{red}{Application}}}
\end{center}
\noindent Déterminer les réels $a$ et $b$ de telle sorte que le polynôme $x^4 + ax^2 + b$ soit divisible par $x^2 - 6x + 5$.

\begin{center}
    \textbf{\underline{\textcolor{red}{Correction}}}
\end{center}

\noindent Posons $P(x) = x^4 + ax^2 + b$ \\
et $q(x) = x^2 - 6x + 5$ \\
$q | P \implies$ Toute racine de $q$ est racine de $P$. \\
Déterminons les racines de $q$. \\
$q(x) = x^2 - 6x + 5$ \\
$\Delta = (-6)^2 - 4(1)(5)$ \\
$\Delta = 36 - 20$ \\
$\Delta = 16 \quad ; \quad \sqrt{\Delta} = 4$ \\
$x_1 = \dfrac{6 - 4}{2} = 1 \quad \text{et} \quad x_2 = \dfrac{6 + 4}{2} = 5$

\vspace{1em}

\noindent $1$ et $5$ sont racines de $q$ donc $1$ et $5$ sont racines de $P$. \\
Alors on a : $\begin{cases} P(1) = 0 \\ P(5) = 0 \end{cases}$ \\
$P(1) = (1)^4 + a(1)^2 + b = 1 + a + b$ \\
$P(5) = (5)^4 + a(5)^2 + b = 625 + 25a + b$

\vspace{1em}

\noindent Ainsi, on a : \\
$\begin{cases} 1 + a + b = 0 \\ 625 + 25a + b = 0 \end{cases} \implies -1 \times \begin{cases} a + b = -1 & (E_1) \\ 25a + b = -625 & (E_2) \end{cases}$

\vspace{0.5em}

\noindent $\begin{cases} -a - b = 1 \\ 25a + b = -625 \end{cases}$ \\
\rule{2.5cm}{0.4pt} \\
$24a + 0 = -624$ \\
$a = \dfrac{-624}{24}$ \\
\fbox{$a = -26$}

\vspace{1em}

\noindent Remplaçons la valeur de $a$ dans $(E_1)$ : \\
$-26 + b = -1$ \\
$b = -1 + 26$ \\
\fbox{$b = 25$}

\vspace{1.5em}

% --- Page 3 : Division euclidienne ---
\subsection*{%
    \color{red}2 - \underline{ Division euclidienne}%
}

\textbf{\underline{\textcolor{red}{Théorème - Définition}}}
  
\noindent Soient $P$ et $q$ deux polynômes. Alors, il existe un unique polynôme $Q$ et un unique polynôme $R$ tels que :
$$P = qQ + R \quad \text{avec} \quad {\color{red} \text{deg}(R) < \text{deg}(q)}$$

\noindent Cette écriture est appelée {\color{red} division euclidienne} de $P$ par $q$. \\
Le polynôme $Q$ est appelé {\color{red} quotient} ; le polynôme $R$ est appelé {\color{red} reste}.

\vspace{1.5em}

% --- Page 4 : Remarque et Application 1 ---
\noindent {\color{red} \underline{Remarque :}} 
\begin{minipage}[t]{0.3\textwidth}
    \begin{tabular}{c|c}
        $P$ & $q$ \\
        \cline{2-2}
        $R$ & $Q$
    \end{tabular}
\end{minipage}
\begin{minipage}[t]{0.6\textwidth}
    $P = qQ + R$ \\
    où $\text{deg}(R) < \text{deg}(q)$
\end{minipage}

\vspace{1em}

\noindent En particulier : \\
$R = 0 \iff P = qQ$ \\
Donc $q$ divise $P$ si $R = 0$.

\vspace{1em}

\begin{center}
    \textbf{\underline{\textcolor{red}{Application 1 }}}
\end{center}
\noindent Soit $f(x) = \frac{2x^4 - x^3 - 2x^2 + 2x - 1}{x^2 - x + 1}$ \\
Déterminons $a, b, c, d$ et $e$ tels que :
$$f(x) = ax^2 + bx + c + \frac{dx + e}{x^2 - x + 1}$$

\begin{center}
    \textbf{\underline{\textcolor{red}{Correction}}}
\end{center}

% Division posée
\begin{center}
\begin{tabular}{r|l}
    $2x^4 - x^3 - 2x^2 + 2x - 1$ & $x^2 - x + 1$ \\
    \cline{2-2}
    $- (2x^4 - 2x^3 + 2x^2) \quad \quad$ & $2x^2 + x - 3$ \\
    \cline{1-1}
    $0 + x^3 - 4x^2 + 2x - 1$ & \\
    $- (x^3 - x^2 + x) \quad$ & \\
    \cline{1-1}
    $0 - 3x^2 + x - 1$ & \\
    $- (-3x^2 + 3x - 3)$ & \\
    \cline{1-1}
    $0 - 2x + 2$ & 
\end{tabular}
\end{center}

\vspace{1em}

\noindent Donc $2x^4 - x^3 - 2x^2 + 2x - 1 = (x^2 - x + 1)(2x^2 + x - 3) + (-2x + 2)$

\vspace{1em}

\noindent $f(x) = \dfrac{(x^2 - x + 1)(2x^2 + x - 3) + (-2x + 2)}{x^2 - x + 1}$

\vspace{1em}

\noindent $f(x) = \dfrac{(x^2 - x + 1)(2x^2 + x - 3)}{x^2 - x + 1} + \dfrac{(-2x + 2)}{x^2 - x + 1}$

\vspace{1em}

\noindent $f(x) = 2x^2 + x - 3 + \dfrac{-2x + 2}{x^2 - x + 1}$

\vspace{1.5em}

\noindent D'où {\color{red} $a = 2 \ ; \ b = 1 \ ; \ c = -3 \ ; \ d = -2 \ \text{et} \ e = 2$}

% --- Application 2 ---
\begin{center}
    \textbf{\underline{\textcolor{red}{Application 2 }}}
\end{center}

\noindent Un polynôme en $x$ divisé séparément par $x-1$ et $x-2$ donne pour restes respectifs $6$ et $18$. \\
Quel reste donnera-t-il si on le divise par $(x-1)(x-2)$ ?

\begin{center}
    \textbf{\underline{\textcolor{red}{Correction}}}
\end{center}

\noindent Soit $P$ le polynôme et $R$ le reste de la division de $P$ par $q(x) = (x-1)(x-2)$. \\
Alors, $P(x) = (x-1)(x-2)Q(x) + R(x)$ où $\text{deg}(R) < \text{deg}(q) = 2$. \\
Déterminons $R$. \\
$\text{deg}(R) < 2 \implies \text{deg}(R) \leq 1 \iff \text{deg}(R) = 0 \text{ ou } \text{deg}(R) = 1$. \\
Donc $R(x) = ax + b$ où $a, b \in \mathbb{R}$. \\
D'où $P(x)$ devient :
$$P(x) = (x-1)(x-2)Q(x) + ax + b$$
$P(1) = (1-1)(1-2)Q(1) + a(1) + b \implies P(1) = a + b$ \\
$P(2) = (2-1)(2-2)Q(2) + a(2) + b \implies P(2) = 2a + b$

\noindent De plus :
$\begin{cases} P(x) = (x-1)Q_1(x) + 6 \\ P(x) = (x-2)Q_2(x) + 18 \end{cases} \implies \begin{cases} P(1) = 6 \\ P(2) = 18 \end{cases}$

\noindent On obtient le système :
$-1 \times \begin{cases} a + b = 6 & (E_1) \\ 2a + b = 18 & (E_2) \end{cases} \implies \begin{cases} -a - b = -6 \\ 2a + b = 18 \end{cases} \implies a = 12$

\noindent Remplaçons la valeur de $a$ dans $(E_1)$ : \\
$12 + b = 6 \implies b = -6$ \\
D'où : \fbox{${\color{blue} R(x) = 12x - 6}$}

\vspace{2em}

% --- Application 3 ---
\begin{center}
    \textbf{\underline{\textcolor{red}{Application 2 }}}
\end{center}

\noindent Déterminer les réels $a$ et $b$ pour que le polynôme $x^4 + ax^2 + bx + 2$ soit divisible par $x^2 + x + 1$.

\begin{center}
    \textbf{\underline{\textcolor{red}{Correction}}}
\end{center}

\noindent $q | P$ si $R = 0$.

\begin{center}
\begin{tabular}{r|l}
    $x^4 \quad + 0x^3 \quad + ax^2 \quad + bx \quad + 2$ & $x^2 + x + 1$ \\
    \cline{2-2}
    $- (x^4 + x^3 + x^2) \quad \quad \quad \quad$ & $x^2 - x + a$ \\
    \cline{1-1}
    $0 - x^3 + (a-1)x^2 + bx + 2$ & \\
    $- (-x^3 - x^2 - x) \quad \quad$ & \\
    \cline{1-1}
    $0 + ax^2 + (b+1)x + 2$ & \\
    $- (ax^2 + ax + a) \quad$ & \\
    \cline{1-1}
    $0 + (b+1-a)x + 2-a$ & 
\end{tabular}
\end{center}

\noindent $R(x) = (b+1-a)x + 2-a$ \\
Donc $R(x) = 0 \iff \begin{cases} b + 1 - a = 0 \\ 2 - a = 0 \end{cases}$ \\
\fbox{${\color{black} a = 2 \text{ et } b = 1}$}
